00159 GLfloat CGLMatrix::calcSarrus(int deletedColumn){
00160     int sourceColumn = 0;
00161     int destColumn = 1;
00162     int row = 1;
00163     GLfloat result;
00164     CGLMatrix dest;   //zero init using standard constructor
00165     //create a 3 by 3 matrix from the top 3 rows, and delete specified column
00166     for(destColumn =1; destColumn <=3; destColumn++) {
00167         sourceColumn++;
00168         if(sourceColumn == deletedColumn)
00169             sourceColumn++;  //skip deleted column
00170         for(row = 1; row <=3; row ++){
00171             dest.setM(destColumn, row, m(sourceColumn, row));
00172         } //end for row
00173     }//end for sourceColumn
00174 
00175     result=
00176             dest.m(1,1) * dest.m(2,2) * dest.m(3,3)       //Sarrus law
00177             + dest.m(1,2) * dest.m(2,3) * dest.m(3,1)
00178             + dest.m(1,3) * dest.m(2,1) * dest.m(3,2)
00179             - dest.m(1,3) * dest.m(2,2) * dest.m(3,1)
00180             - dest.m(1,1) * dest.m(2,3) * dest.m(3,2)
00181             - dest.m(1,2) * dest.m(2,1) * dest.m(3,3);
00182     QString s;
