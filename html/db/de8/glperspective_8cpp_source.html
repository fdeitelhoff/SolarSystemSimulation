00041 void GLPerspective::apply(){
00042   glMatrixMode(GL_PROJECTION); //switch to projection matrix
00043   glLoadIdentity();            //initialize matrix with identity matrix
00044   //multipy perspective transformation on projection matrix
00045   gluPerspective(_Fovy,    //fovy
00046                  _Aspect, //aspect
00047                  _Near,    //near
00048                  _Far);  //far
00049 
00050   glMatrixMode(GL_MODELVIEW);
00051   glLoadIdentity();
00052   gluLookAt(_Camera.x(), _Camera.y(), _Camera.z(), //eye
00053             _Center.x(), _Center.y(), _Center.z(),  //center
00054             _Up.x(), _Up.y(), _Up.z()); //up
00055 }
