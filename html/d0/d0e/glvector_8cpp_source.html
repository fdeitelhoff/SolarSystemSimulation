00163 //****************Function RotateVector************************************
00164 //First get the projection of v1 on rotation axis. Then make a 2d axis
00165 //system where Rotation.axis is normal to. xNew is the first axis of this
00166 //system xNew = v1 - Av1(Projection of v1 on Rotation.Axis). The second axis
00167 //yNew is obtained by vectormultiplication of Av1 and xNew divided by |Av1|.
00168 //this has the same length as xNew and is normal to xNew and Rotation.Axis.
00169 //Now xNew can easily be rotated in a 2d coordinate system made by xNew and
00170 //yNew. The result is xRotated. The result of rotating v1 is Av1 + xRotated.
00171 
00172 const GLVector GLVector::rotateVector(const GLVector& Axis, GLdouble Angle)const{
00173 GLVector vAxis = Axis; //local copy
00174 GLVector Av1     ; //Projection of v1 on Rotation Axis
00175 GLVector xNew    ; //v1-Av1 first vector in plane of rotation
00176 GLVector yNew    ; //second vector in plane of rotation
00177 GLVector xRotated; //result of rotation of xnew by Rotation.Theta
00178 GLVector xNewComp; //xNew component of xRotated in xNew, yNew plane
00179 GLVector yNewComp; //yNew component of xRotated in xNew, yNew plane
00180 GLdouble     AbsAxis;    // |Rotation Axis|
00181 GLdouble     AbsAv1 ;    // |Av1|
00182 GLdouble   radAngle = toRadian(Angle);
00183  // precalculate lengths
00184  AbsAxis = vAxis.length();  // precalculate length of Axis
00185  AbsAv1  = (*this * vAxis )/ AbsAxis ;   // precalculate length of Av1
00186 
00187  //get projection of v1 on Axis
00188  Av1     = vAxis * AbsAv1 ;
00189 
00190  // Make a cartesian xNew,yNew plane with equal xNew and yNew lengths.
00191  // Because xNew is normal to Axis, sin(Axis, xNew) is 1
00192  xNew    = * this - Av1; //keep length != 1 for speed
00193  yNew    = vAxis.vectorMult(xNew) * ( 1.0 / AbsAxis );
00194 
00195  // Rotate xNew in xNew, yNew plane by Theta
00196  xNewComp= xNew * cos( radAngle );
00197  yNewComp= yNew * sin( radAngle );
00198  xRotated= xNewComp + yNewComp;
00199 
00200  //Add Av1 and xRotated to get final result
00201  return Av1 + xRotated;
00202 }
