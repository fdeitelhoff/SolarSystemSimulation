\-Establish database connection. \-On windows systems use \-O\-D\-B\-C, on other systems the native \-Postgre\-S\-Q\-L driver. 


\begin{DoxyCode}
{
    #ifdef WIN32
    // Connection over ODBC!. "SolarSystemSimulation" is the name of the ODBC
       data source.
    database = QSqlDatabase::addDatabase("QODBC");
    database.setDatabaseName("SolarSystemSimulation");

    #else
    // Connection through the native PostgreSQL driver.
    database = QSqlDatabase::addDatabase("QPSQL");
    database.setHostName("localhost");
    database.setDatabaseName("SolarSystemSimulation");
    database.setUserName("postgres");
    database.setPassword("postgres");
    #endif

    if (!database.open())
    {
        throw DatabaseConnectionFailedException("The database connection could
       not be opened!",
                                                database.lastError().text());
    }
}
